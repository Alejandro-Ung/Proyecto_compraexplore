	captura;
FinProceso

SubProceso captura 
	Definir N1, N2 como entero;
	Limpiar Pantalla;
	Escribir "Calculadora";
	Escribir " ";
	Escribir 'Ingrese Primer Valor: ';
	Leer N1;
	Escribir 'Ingrese Segundo Valor: ';
	Leer N2;
	Escribir " ";
	Menu(N1,N2);
FinSubProceso

SubProceso Menu(N1 Por Valor,N2 Por Valor)
	Definir Sel1 Como Entero;
	Limpiar Pantalla;
	Escribir "Que desea hacer";
	Escribir "1) Suma";
	Escribir "2) Resta";
	Escribir "3) División";
	Leer Sel1;
	Segun Sel1 hacer
		1:Suma(N1,N2);
		2:Resta(N1,N2);
		3:Div(N1,N2);
	FinSegun
	
FinSubProceso

SubAlgoritmo Suma (N1 Por Valor,N2 Por Valor)
	Limpiar Pantalla;
	Escribir 'La sumatoria de los numeros es: ', N1+N2;
	Escribir "--------------------------------------------------------";
	Volver(N1,N2);
FinSubAlgoritmo

SubAlgoritmo Resta (N1 Por Valor,N2 Por Valor)
	Limpiar Pantalla;
	Escribir 'La resta de los numeros es: ', N1-N2;
	Escribir " ";
	Volver(N1,N2);
FinSubAlgoritmo

SubAlgoritmo Div (N1 Por Valor,N2 Por Valor)
	Limpiar Pantalla;
	Si N2=0 Entonces
		Escribir 'El valor no puede ser 0';
	SiNo
		Escribir 'El cociente de los numeros es: ', N1/N2;
	FinSi
	Escribir " ";
	Volver(N1,N2);
FinSubAlgoritmo

SubAlgoritmo Volver(N1,N2)
	Definir Sel2 Como Entero;
	Escribir "Seleccione segun lo que usted desee";
	Escribir " ";
	Escribir " 1) Si desea volver al Menú";
	Escribir " 2) Si desea hacer operaciones con otros Numeros";
	Escribir " 3) Si desea Salir";
	Escribir " ";
	Leer Sel2;
	Si Sel2=1 Entonces
		Menu(N1,N2);
	SiNo
		Si Sel2=3 Entonces
			Salir;
		Sino 
			si Sel2=2 Entonces
				captura;
			SiNo
				Escribir "Fin de la sesión";
			FinSi
		FinSi
	FinSi
FinSubAlgoritmo

SubProceso Salir
	Limpiar Pantalla;
	Escribir "Gracias Por Vivir la experiencia ReyesCalculable";
FinSubProceso
